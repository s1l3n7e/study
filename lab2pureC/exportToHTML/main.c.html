#include <stdio.h> 
#include <math.h> 
#include <stdlib.h> 
#include <windows.h> 
 
void first() 
{ 
    int 
            size = 0, 
            i, j; 
 
    printf("Введите размер квадрата\n"); 
    scanf("%d", &size); 
    for (i = 0; i < size; i++) 
    { 
        for (j = 0; j < size; j++) 
        { 
            printf("*"); 
        } 
        printf("\n"); 
    } 
} 
 
void second() 
{ 
    int 
            number = 0, 
            result = 0, 
            i; 
 
    printf("Введите число в двоичной СС\n"); 
    scanf("%d", &number); 
    for(i = 0; number != 0; i++) 
    { 
        result += number % 10 * pow(2, i); 
        number /= 10; 
    } 
    printf("%d", result); 
}
